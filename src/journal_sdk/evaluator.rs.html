73pub struct Evaluator {
74    pub sc: *mut s7_scheme,
75    primitives: Vec<Primitive>,
76}
77
78impl Evaluator {
79    pub fn new(types: HashMap<i64, Type>, primitives: Vec<Primitive>) -> Self {
80        let mut primitives_ = vec![
81            primitive_hex_string_to_byte_vector(),
82            primitive_byte_vector_to_hex_string(),
83            primitive_expression_to_byte_vector(),
84            primitive_byte_vector_to_expression(),
85        ];
86
87        primitives_.extend(primitives);
88
89        unsafe {
90            let sc: *mut s7_scheme = s7_init();
91
92            // remove insecure primitives
93            for primitive in REMOVE {
94                s7_define(
95                    sc,
96                    s7_rootlet(sc),
97                    s7_make_symbol(sc, primitive.as_ptr()),
98                    s7_make_symbol(sc, c"*removed*".as_ptr())
99                );
100            }
101
102            // add new types
103            for (&tag_, type_) in types.iter() {
104                let tag = s7_make_c_type(sc, type_.name.as_ptr());
105                assert!(tag == tag_, "Type tag was not properly set");
106                s7_c_type_set_gc_free(sc, tag, Some(type_.free));
107                s7_c_type_set_gc_mark(sc, tag, Some(type_.mark));
108                s7_c_type_set_is_equal(sc, tag, Some(type_.is_equal));
109                s7_c_type_set_to_string(sc, tag, Some(type_.to_string));
110            }
111
112            // add new primitives
113            for primitive in primitives_.iter() {
114                s7_define_function(
115                    sc,
116                    primitive.name.as_ptr(),
117                    Some(primitive.code),
118                    primitive.args_required.try_into().unwrap(),
119                    primitive.args_optional.try_into().unwrap(),
120                    primitive.args_rest,
121                    primitive.description.as_ptr(),
122                );
123            }
124
125            Self { sc, primitives: primitives_ }
126        }
127
128    }
129
130    pub fn evaluate(&self, code: &str) -> String {
131        unsafe {
132            // execute query and return
133            let wrapped = CString::new(format!(
134                "(catch #t (lambda () (eval (read (open-input-string \"{}\")))) (lambda x {}))",
135                code.replace("\\", "\\\\").replace("\"", "\\\""),
136                "`(error ',(car x) ,(apply format (cons #f (cadr x))))",
137            )).unwrap();
138            let s7_obj = s7_eval_c_string(self.sc, wrapped.as_ptr());
139            let output = s7_object_to_c_string(self.sc, s7_obj);
140            let ret = CStr::from_ptr(output).to_str().unwrap().to_owned();
141            free(output as *mut libc::c_void);
142            ret
143        }
144    }
145}

155fn primitive_expression_to_byte_vector() -> Primitive {
156    unsafe extern "C" fn code(sc: *mut s7_scheme, args: s7_pointer) -> s7_pointer {
157        let arg = s7_car(args);
158
159        let s7_c_str = s7_object_to_c_string(sc, arg);
160        let c_string = CStr::from_ptr(s7_c_str);
161
162        let bv = s7_make_byte_vector(sc, c_string.to_bytes().len() as i64, 1 as i64, std::ptr::null_mut());
163        for (i, b) in c_string.to_bytes().iter().enumerate() { s7_byte_vector_set(bv, i as i64, *b); }
164        free(s7_c_str as *mut libc::c_void);
165        bv
166    }
167
168    Primitive::new(
169        code,
170        c"expression->byte-vector",
171        c"(expression->byte-vector expr) convert a expression string to a byte vector",
172        1, 0, false,
173    )
174}

176fn primitive_byte_vector_to_expression() -> Primitive {
177    unsafe extern "C" fn code(sc: *mut s7_scheme, args: s7_pointer) -> s7_pointer {
178        let arg = s7_car(args);
179
180        if !s7_is_byte_vector(arg) {
181            return s7_wrong_type_arg_error(
182                sc, c"byte-vector->expression".as_ptr(), 1, arg,
183                c"a byte-vector".as_ptr())
184        }
185
186        let mut bytes = vec![39];  // quote so that it evaluates correctly
187        for i in 0..s7_vector_length(arg) { bytes.push(s7_byte_vector_ref(arg, i)) }
188        bytes.push(0);
189
190        let c_string = CString::from_vec_with_nul(bytes).unwrap();
191        s7_eval_c_string(sc, c_string.as_ptr())
192    }
193
194    Primitive::new(
195        code,
196        c"byte-vector->expression",
197        c"(byte-vector->expression bv) convert a byte vector to an expression",
198        1, 0, false,
199    )
200}

202fn primitive_hex_string_to_byte_vector() -> Primitive {
203    unsafe extern "C" fn code(sc: *mut s7_scheme, args: s7_pointer) -> s7_pointer {
204        let arg = s7_car(args);
205
206        if !s7_is_string(arg) {
207            return s7_wrong_type_arg_error(
208                sc, c"hex-string->byte-vector".as_ptr(), 1, arg,
209                c"a hex string".as_ptr())
210        }
211
212        let s7_c_str = s7_object_to_c_string(sc, arg);
213        let hex_string = CStr::from_ptr(s7_c_str).to_str().unwrap();
214
215        let result: Result<Vec<u8>, ParseIntError> = (1..hex_string.len()-1)
216            .step_by(2)
217            .map(|i| u8::from_str_radix(&hex_string[i..i + 2], 16))
218            .collect();
219
220        free(s7_c_str as *mut libc::c_void);
221
222        match result {
223            Ok(result) => {
224                let bv = s7_make_byte_vector(sc, result.len() as i64, 1 as i64, std::ptr::null_mut());
225                for i in 0..result.len() { s7_byte_vector_set(bv, i as i64, result[i]); }
226                bv
227            }
228            _ => {
229                s7_wrong_type_arg_error(
230                    sc, c"hex-string->byte-vector".as_ptr(), 1, arg,
231                    c"a hex string".as_ptr())
232            }
233        }
234    }
235
236    Primitive::new(
237        code,
238        c"hex-string->byte-vector",
239        c"(hex-string->byte-vector str) convert a hex string to a byte vector",
240        1, 0, false,
241    )
242}

244fn primitive_byte_vector_to_hex_string() -> Primitive {
245    unsafe extern "C" fn code(sc: *mut s7_scheme, args: s7_pointer) -> s7_pointer {
246        let arg = s7_car(args);
247
248        if !s7_is_byte_vector(arg) {
249            return s7_wrong_type_arg_error(
250                sc, c"byte-vector->hex-string".as_ptr(), 1, arg,
251                c"a byte-vector".as_ptr())
252        }
253
254        let mut bytes = vec![0 as u8; s7_vector_length(arg) as usize];
255        for i in 0..bytes.len() as usize { bytes[i] = s7_byte_vector_ref(arg, i as i64); }
256
257        let mut string = String::with_capacity(bytes.len() * 2);
258        for b in bytes { write!(&mut string,"{:02x}", b).unwrap(); }
259
260        // todo: this might cause a pointer issue
261        let c_string = CString::new(string).unwrap();
262        s7_object_to_string(sc, s7_make_string(sc, c_string.as_ptr()), false)
263    }
264
265    Primitive::new(
266        code,
267        c"byte-vector->hex-string",
268        c"(byte-vector->hex-string bv) convert a byte vector to a hex string",
269        1, 0, false,
270    )
271}
